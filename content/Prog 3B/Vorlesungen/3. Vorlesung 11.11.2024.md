## Vererbung

Wir implementieren erst Kamera, da diese aus dem URL Diagramm einen Block bildet und viele von dieser erben.
```cpp
#pragma once

class Camera
{
protected: // wie private aber andere Klassen können drauf zugreifen
	std::string make;
	std::string model
	unsigned int year;

public:
	Camera();// default konstruktor
	Camera(std::string make, std::string model, unsigned int year); // initialiserit Attribute

	virtual void takePhoto(); //soll überschrieben werden können
	
};
```

digitalcamera.h soll von camera.h erben

```cpp
#pragma once
#include "Camera.h"

class DigitalCamera : public Camera
//public nimmt Methoden/Attribute und behält deren Sichtbarkeit bei
{
private:
	unsigned int megapixel;

public:
	DigitalCamera(); // deafult Constructor
	DigitalCamera(unsigned int year, std::string make, std::string model, unsigned int megapixel)

	void takePhoto() override;
};
```

filmcamera.h
```cpp
#pragma once
#include "Camera.h"

class FilmCamera : public Camera
{
private:
	std::string film;

public:
	FilmCamera(); // deafult Constructor
	FilmCamera(unsigned int year, std::string make, std::string model, std::string film)

	void takePhoto() override;
};
```

digitalcamera.cpp
```cpp
#include <digitalcamera.h>

DigitalCamera::DigitalCamera() : Camera()
{}

DigitalCamera::DigitalCamera(unsigned int year, std::string make, std::string model, unsigned int megapixel) : Camera(make, model, year), megapixel(megapixel)
{}

void DigitalCamera::takePhoto()
{
	cout << "Taking photo with " << megapixel << " digital camera "  << make << " " << model << endl;
}
```
